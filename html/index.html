VT SDK  1.0.0
API for controlling the headset state
Tutorial

Below is the an example of the SDK usage 

            #include <sdk.h>
            #include <iostream>
            using namespace std;

            int main()
            {
                //init the SDK
                auto res = VtSDK::init();
                if (res == 0) {
                  //get a list of devices attached to the PC 
                  auto devices = VtSDK::enumerateDevices();

                  //use the first one for call control, just for demo purpose
                  auto device = devices[0];

                  //listen to the device event(i.e. user interaction on the physical buttons on the device)
                  bool success = device->subscribe([](DeviceEvent ev) {
                      if (ev == DeviceEvent::OFFHOOK) {
                          cout << "deive has gone to offhook state, softphone should answer the call"<<endl;
                          .....softphone sync the state 
                      }
                      if (ev == DeviceEvent::ONHOOK) {
                          cout << "deive has gone to onhook state, softphone should end the call" <<endl;
                          .....softphone sync the state 
                      }
                      if (ev == DeviceEvent::MUTE) {
                          cout << "deive mute has toggled" << endl;
                          .....softphone sync the state 
                      }
                      if (ev == DeviceEvent::REJECT) {
                          cout << "incomming call was rejected by the device" << endl;
                          .....softphone sync the state 
                      }
                  });
                  if (success) {
                      //start ringing effect on the device
                      device->ring();
                  }
                  while (true)
                  {
                     ...softphone normal event loop
                  }
                  //exit the sdk, should be called whenever possible
                  VtSDK::exit();
                }
                return 0;
            }
